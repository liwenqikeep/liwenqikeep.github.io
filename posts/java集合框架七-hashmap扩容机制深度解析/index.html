иҙҹиҪҪеӣ еӯҗзҡ„еҪұе“Қ

иҙҹиҪҪеӣ еӯҗ жү©е®№йҳҲеҖј з©әй—ҙеҲ©з”ЁзҺҮ жҹҘжүҫ/жҸ’е…ҘжҖ§иғҪ
0.5 capacity * 0.5 дҪҺпјҲжөӘиҙ№з©әй—ҙпјү еҝ«пјҲеҶІзӘҒе°‘пјү
0.75 capacity * 0.75 е№іиЎЎпјҲжҺЁиҚҗпјү е№іиЎЎ
1.0 capacity * 1.0 й«ҳпјҲзңҒз©әй—ҙпјү ж…ўпјҲеҶІзӘҒеӨҡпјү

з»“и®әпјҡй»ҳи®Өзҡ„ 0.75 жҳҜж—¶й—ҙе’Ңз©әй—ҙзҡ„е№іиЎЎзӮ№пјҢHashMap дҪңиҖ…зҡ„з»ҸйӘҢеҖјгҖӮ

жү©е®№и®Ўз®—

resize() ж–№жі•

final Node<K,V>[] resize() {
    Node<K,V>[] oldTab = table;
    int oldCap = (oldTab == null) ? 0 : oldTab.length;
    int oldThr = threshold;
    
    int newCap, newThr = 0;
    
    if (oldCap > 0) {
        // 1. е·ІиҫҫжңҖеӨ§е®№йҮҸпјҢдёҚеҶҚжү©е®№
        if (oldCap >= MAXIMUM_CAPACITY) {
            threshold = Integer.MAX_VALUE;
            return oldTab;
        }
        // 2. ж–°е®№йҮҸ = ж—§е®№йҮҸ * 2
        else if ((newCap = oldCap << 1) < MAXIMUM_CAPACITY &&
                 oldCap >= DEFAULT_INITIAL_CAPACITY) {
            newThr = oldThr << 1;  // ж–°йҳҲеҖј = ж—§йҳҲеҖј * 2
        }
    }
    else if (oldThr > 0) {
        // 3. з¬¬дёҖж¬Ў putпјҢдҪҝз”Ёжһ„йҖ ж—¶зҡ„еҲқе§Ӣе®№йҮҸ
        newCap = oldThr;
    }
    else {
        // 4. дҪҝз”Ёй»ҳи®ӨеҖј
        newCap = DEFAULT_INITIAL_CAPACITY;  // 16
        newThr = (int)(DEFAULT_LOAD_FACTOR * DEFAULT_INITIAL_CAPACITY);  // 12
    }
    
    // и®Ўз®—ж–°зҡ„йҳҲеҖј
    if (newThr == 0) {
        float ft = (float)newCap * loadFactor;
        newThr = (newCap < MAXIMUM_CAPACITY && ft < (float)MAXIMUM_CAPACITY ?
                  (int)ft : Integer.MAX_VALUE);
    }
    
    threshold = newThr;
    
    // еҲӣе»әж–°ж•°з»„
    @SuppressWarnings({"rawtypes","unchecked"})
    Node<K,V>[] newTab = (Node<K,V>[])new Node[newCap];
    table = newTab;
    
    // иҝҒз§»е…ғзҙ 
    if (oldTab != null) {
        for (int j = 0; j < oldCap; ++j) {
            Node<K,V> e;
            if ((e = oldTab[j]) != null) {
                oldTab[j] = null;
                if (e.next == null) {
                    // еҚ•дёӘиҠӮзӮ№пјҢзӣҙжҺҘз§»еҠЁ
                    newTab[e.hash & (newCap - 1)] = e;
                }
                else if (e instanceof TreeNode) {
                    // зәўй»‘ж ‘иҝҒз§»
                    ((TreeNode<K,V>)e).split(this, newTab, j, oldCap);
                }
                else {
                    // й“ҫиЎЁиҝҒз§»пјҲйҮҚзӮ№пјҒпјү
                    // дјҳеҢ–пјҡеҲҶдёәй«ҳдҪҚй“ҫе’ҢдҪҺдҪҚй“ҫ
                    ...
                }
            }
        }
    }
    return newTab;
}

// JDK 1.7 ж—§ж•°з»„й•ҝеәҰ

// иҝҒз§»й“ҫиЎЁ
void transfer(Entry[] newTable) {
    Entry[] src = table;
    int newCapacity = newTable.length;
    
    for (int j = 0; j < src.length; j++) {
        Entry<K,V> e = src[j];
        if (e != null) {
            src[j] = null;  // йҮҠж”ҫж—§иЎЁ
            
            do {
                Entry<K,V> next = e.next;
                // йҮҚж–°и®Ўз®—зҙўеј•
                int i = indexFor(e.hash, newCapacity);
                e.next = newTable[i];
                newTable[i] = e;
                e = next;
            } while (e != null);
        }
    }
}

HashMap еңЁ JDK 1.7 ж—¶пјҢжү©е®№ж“ҚдҪңдёҚжҳҜзәҝзЁӢе®үе…Ёзҡ„гҖӮеңЁеӨҡзәҝзЁӢзҺҜеўғдёӢпјҢеҸҜиғҪеҜјиҮҙжӯ»еҫӘзҺҜгҖӮ

// JDK 1.7 иҝҒз§»д»Јз Ғз®ҖеҢ–
void transfer(Entry[] newTable) {
    Entry[] src = table;
    int newCapacity = newTable.length;
    
    for (int j = 0; j < src.length; j++) {
        Entry<K,V> e = src[j];
        if (e != null) {
            do {
                Entry<K,V> next = e.next;
                int i = indexFor(e.hash, newCapacity);
                e.next = newTable[i];  // еӨҙжҸ’жі•пјҒ
                newTable[i] = e;
                e = next;
            } while (e != null);
        }
    }
}

жӯ»еҫӘзҺҜжҲҗеӣ еӣҫи§Ј

еҲқе§ӢзҠ¶жҖҒпјҲеҒҮи®ҫдёӨдёӘзәҝзЁӢеҗҢж—¶жү©е®№пјүпјҡ
зәҝзЁӢ A е’Ң B йғҪзңӢеҲ°еҗҢж ·зҡ„й“ҫиЎЁз»“жһ„

еҺҹй“ҫиЎЁпјҡ A вҶ’ B вҶ’ null
        вҶ“
      e=B, next=A
      
зәҝзЁӢ A жү§иЎҢ transfer()пјҢеӨҙжҸ’жі•пјҡ

з¬¬дёҖж¬Ўпјҡe=B
        B.next = newTable[i]
        newTable[i] = B
        з»“жһңпјҡnewTable[i] = B вҶ’ null

з¬¬дәҢж¬Ўпјҡe=A
        A.next = newTable[i] = B
        newTable[i] = A
        з»“жһңпјҡnewTable[i] = A вҶ’ B вҶ’ null

дҪҶеҰӮжһңзәҝзЁӢ A иў«жҢӮиө·пјҢзәҝзЁӢ B е…Ҳе®ҢжҲҗпјҡ

зәҝзЁӢ B е®ҢжҲҗеҗҺзҡ„з»“жһ„пјҡ
newTable[i] = A вҶ’ B вҶ’ null

зәҝзЁӢ A жҒўеӨҚпјҢз»§з»ӯжү§иЎҢ transfer()пјҡ

з¬¬дёҖж¬Ўпјҡe=BпјҲд»Һ src[j] иҜ»еҸ–пјү
        B.next = newTable[i] = A вҶ’ B
        newTable[i] = B
        з»“жһңпјҡnewTable[i] = B вҶ’ A вҶ’ B вҶ’ ...пјҲеҫӘзҺҜпјҒпјү
        
з¬¬дәҢж¬Ўпјҡe=AпјҲnext = Bпјү
        A.next = newTable[i] = B вҶ’ A вҶ’ B ...
        newTable[i] = A
        з»“жһңпјҡnewTable[i] = A вҶ’ B вҶ’ A вҶ’ B вҶ’ ...пјҲжӯ»еҫӘзҺҜпјҒпјү

// йў„дј°е…ғзҙ ж•°йҮҸ nпјҢеҸҚжҺЁ HashMap е®№йҮҸ
// зӣ®ж Үпјҡsize < threshold = capacity * loadFactor
// capacity = n / loadFactor * 1.2пјҲз•ҷ 20% дҪҷйҮҸпјү
// з„¶еҗҺеҗ‘дёҠеҸ–ж•ҙеҲ° 2 зҡ„е№Ӯж¬Ў

int n = 100000;
int capacity = (int) (n / 0.75 * 1.2);  // 160000
// еҗ‘дёҠеҸ–ж•ҙеҲ° 2 зҡ„е№Ӯж¬Ў
int size = 1;
while (size < capacity) {
    size <<= 1;
}
// size = 262144 = 2^18

// еҲ©з”Ё LinkedHashMap е®һзҺ° LRU зј“еӯҳ
public class LRUCache<K, V> extends LinkedHashMap<K, V> {
    private final int maxSize;
    
    public LRUCache(int maxSize) {
        super(16, 0.75f, true);  // accessOrder = true
        this.maxSize = maxSize;
    }
    
    @Override
    protected boolean removeEldestEntry(Map.Entry<K, V> eldest) {
        return size() > maxSize;
    }
}

// дҪҝз”Ё
LRUCache<String, String> cache = new LRUCache<>(1000);
cache.put("key", "value");  // иҮӘеҠЁж·ҳжұ°жңҖиҖҒзҡ„жқЎзӣ®

еёёи§ҒйқўиҜ•йўҳ

Q1пјҡHashMap зҡ„жү©е®№жңәеҲ¶пјҹ

зӯ”пјҡ

  1. еҪ“ size > capacity * loadFactor ж—¶и§ҰеҸ‘
  2. еҲӣе»әж–°ж•°з»„пјҢе®№йҮҸе’ҢйҳҲеҖјйғҪзҝ»еҖҚ
  3. е°ҶеҺҹж•°з»„зҡ„е…ғзҙ иҝҒз§»еҲ°ж–°ж•°з»„
  4. JDK 1.8 дјҳеҢ–пјҡй«ҳдҪҺдҪҚй“ҫиЎЁпјҢйҒҝе…ҚйҮҚж–°и®Ўз®— hash

Q2пјҡJDK 1.8 еҜ№жү©е®№еҒҡдәҶе“ӘдәӣдјҳеҢ–пјҹ

зӯ”пјҡ

  1. еҲ©з”Ё (e.hash & oldCap) еҲӨж–ӯжҳҜеҗҰйңҖиҰҒз§»еҠЁеҲ°й«ҳдҪҚпјҢйҒҝе…ҚйҮҚж–°и®Ўз®— hash
  2. е°Ҷй“ҫиЎЁеҲҶдёәй«ҳдҪҺдҪҚдёӨдёӘй“ҫиЎЁпјҢеӨҙжҸ’жі•ж”№дёәе°ҫжҸ’жі•йЎәеәҸдёҚеҸҳ
  3. зәўй»‘ж ‘жңүдё“й—Ёзҡ„ split() ж–№жі•еӨ„зҗҶ

Q3пјҡHashMap еңЁ JDK 1.7 еӨҡзәҝзЁӢдёӢдјҡжӯ»еҫӘзҺҜпјҹ

зӯ”пјҡжҳҜзҡ„гҖӮJDK 1.7 зҡ„жү©е®№йҮҮз”ЁеӨҙжҸ’жі•пјҢеӨҡзәҝзЁӢдёӢеҸҜиғҪеҜјиҮҙй“ҫиЎЁжҲҗзҺҜпјҢеҪўжҲҗжӯ»еҫӘзҺҜгҖӮJDK 1.8 е·Ідҝ®еӨҚпјҲдҪҶд»ҚдёҚжҳҜзәҝзЁӢе®үе…Ёзҡ„пјүгҖӮ

зӯ”пјҡдҫҝдәҺз”Ё (n-1) & hash еҝ«йҖҹи®Ўз®—зҙўеј•пјҢиҖҢдё”жү©е®№ж—¶еҸҜд»Ҙз”Ё (e.hash & oldCap) еҲӨж–ӯй«ҳдҪҚпјҢз®ҖеҢ–иҝҒз§»йҖ»иҫ‘гҖӮ

  4. еӨҡзәҝзЁӢй—®йўҳпјҡJDK 1.7 еҸҜиғҪжӯ»еҫӘзҺҜпјҢJDK 1.8 дёҚдјҡдҪҶд»ҚйқһзәҝзЁӢе®үе…Ё
